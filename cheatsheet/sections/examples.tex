\section{Examples}

\subsection{Converting to LP Standard Form}
\textbf{Problem}: Given $\alpha_1,\ldots,\alpha_n$,
$\min_{x \in \R^n} e^{(\sum_{i=1}^n \alpha_i x_i)}$ s.t.
$(\|x\|_\infty + 10 \|x\|_1)^2 \leq 5$.

\textbf{Step 1}: $\min e^y \Leftrightarrow \min y$ (strictly
increasing). Let $y = \sum_i \alpha_i x_i$. Constraint:
$\|x\|_\infty + 10\|x\|_1 \leq \sqrt{5}$.

\textbf{Step 2}: Linearize norms. $\|x\|_\infty \leq s
\Leftrightarrow -x_i \leq s, x_i \leq s$ for all $i$.
$\|x\|_1 \leq t \Leftrightarrow \|x\|_1 \leq \sum_{i=1}^n
r_i$ where $|x_i| \leq r_i \Leftrightarrow -x_i \leq r_i,
x_i \leq r_i$. Constraint becomes $s + 10t \leq \sqrt{5}$,
$\sum_i r_i \leq t$, $x_i \leq s$, $-x_i \leq s$,
$x_i \leq r_i$, $-x_i \leq r_i$ for all $i$.

\textbf{Step 3}: Eliminate $t$. Since $s,r$ already upper
bounds, set $t = \sum_i r_i$. New constraint:
$s + 10\sum_i r_i \leq \sqrt{5}$.

\textbf{LP form}: $\min_{\substack{x,r,s}} \alpha^\top x$
s.t. $s + 10 \cdot \mathbbm{1}_n^\top r = \sqrt{5}$,
$s - x_i \geq 0$, $s + x_i \geq 0$, $r_i - x_i \geq 0$,
$r_i + x_i \geq 0$ for $i=1,\ldots,n$ (where
$\alpha = (\alpha_1,\ldots,\alpha_n)$ and
$r = (r_1,\ldots,r_n)$).

\textbf{Standard form}: $\min_{x^+,x^-,r,s,u,v,w,z}
\alpha^\top(x^+ - x^-)$ s.t.
$s + 10 \cdot \mathbbm{1}_n^\top r = \sqrt{5}$,
$r_i - (x_i^+ - x_i^-) - u_i = 0$,
$r_i + (x_i^+ - x_i^-) - v_i = 0$,
$s - (x_i^+ - x_i^-) - w_i = 0$,
$s + (x_i^+ - x_i^-) - z_i = 0$ for $i=1,\ldots,n$, with
$x^+,x^-,r,s,u,v,w,z \geq 0$.

\subsection{Proving Emptiness via Weak Duality}
\textbf{Problem}: Prove $\mathcal{S} = \{x \in \R^2 \mid
x_1^2 + x_2^2 = 1, x_1 + x_2 \geq 2\}$ is empty.

\textbf{Solution}: Consider primal $\min_{x \in \R^2} 0$
s.t. $x_1^2 + x_2^2 = 1$, $x_1 + x_2 - 2 \geq 0$.
Lagrangian: $L(x,\mu,\lambda) = \mu(x_1^2 + x_2^2 - 1) +
\lambda(2 - x_1 - x_2)$.

Dual function:
$d(\mu,\lambda) = \begin{cases}
L\left(\frac{\lambda}{2\mu}\mathbbm{1}, \mu, \lambda\right)
& \mu > 0 \\
-\infty & \mu < 0 \\
-\infty & \mu = 0, \lambda \neq 0 \\
0 & \mu = \lambda = 0
\end{cases}$

For $\mu > 0$: $d(\mu,\lambda) = \frac{\lambda^2}{4\mu} +
\mu - 2\lambda$. By weak duality, $d(\mu,\lambda) \leq p^*$
for all $\mu,\lambda \geq 0$. If $\exists \mu > 0, \lambda
\geq 0$ s.t. $d(\mu,\lambda) > 0$, then $p^* > 0$, implying
$\mathcal{S}$ is empty.

Find $\mu,\lambda$ s.t. $\frac{\lambda^2}{4\mu} + \mu -
2\lambda > 0 \Rightarrow \lambda^2 - 4\lambda\mu + 4\mu^2 <
0 \Rightarrow (\lambda - 2\mu)^2 < 2\mu^2 \Rightarrow
(2 - \sqrt{2})\mu < \lambda < (2 + \sqrt{2})\mu$.

Choose any $\mu > 0$ and $\lambda = 2\mu$ (satisfies
inequality) to obtain certificate.

\subsection{Solving via Strong Duality}
\textbf{Problem}: Given $A \in \R^{n \times n}$ symmetric
with $A \succ 0$ and $c \in \R^n$ with $c \neq 0$, find
optimal value of $\min_{x \in \R^n} c^\top x$ s.t.
$x^\top A x \leq 1$.

\textbf{Solution}: Lagrangian:
$L(x,\lambda) = c^\top x + \lambda(x^\top A x - 1)$.

\textbf{Dual function}: Minimize over $x$. Set
$\nabla_x L(x,\lambda) = c + 2\lambda A x = 0 \Rightarrow
x = -\frac{1}{2\lambda} A^{-1} c$ (requires $A$ PD).

Substitute back:
$L(x,\lambda) = c^\top x + \lambda x^\top A x - \lambda =
-\frac{1}{2\lambda} c^\top A^{-1} c + \lambda \left(
\frac{1}{4\lambda^2} c^\top A^{-1} c \right) - \lambda =
-\frac{1}{4\lambda} c^\top A^{-1} c - \lambda$.

Thus $\phi(\lambda) = -\lambda - \frac{1}{4\lambda} c^\top
A^{-1} c$ for $\lambda > 0$.

\textbf{Dual problem}: $\max_{\lambda \geq 0} \phi(\lambda) =
\max_{\lambda > 0} -\lambda - \frac{1}{4\lambda} c^\top
A^{-1} c$.

\textbf{Solve dual}: Let $s = c^\top A^{-1} c > 0$. Since
convex and Slater's condition holds, strong duality applies.
Minimize $-\phi(\lambda) = \lambda + \frac{s}{4\lambda}$.
Take derivative: $1 - \frac{s}{4\lambda^2} = 0 \Rightarrow
\lambda^* = \frac{\sqrt{s}}{2}$.

Substitute: $\phi(\lambda^*) = -\frac{\sqrt{s}}{2} -
\frac{s}{4 \cdot \frac{\sqrt{s}}{2}} = -\frac{\sqrt{s}}{2} -
\frac{\sqrt{s}}{2} = -\sqrt{s}$.

\textbf{Optimal value}: $p^* = -\sqrt{c^\top A^{-1} c}$.

\subsection{KKT \& Constraint Sensitivity}
\textbf{Problem}: $\min_{x \in \R^2} x_1 + x_2$ s.t.
$(x_1-1)^2 + (x_2-2)^2 \leq 1$, $x_1^2 + x_2^2 \leq 100$.

\textbf{Classification}: Linear objective (convex). Both
constraints are quadratic with PSD Hessians, hence convex.
This is a \textbf{QCQP} (Quadratically Constrained Quadratic
Program).

\textbf{Slater's condition}: Holds. Point $(1,2)$ satisfies
both constraints strictly: $(1-1)^2 + (2-2)^2 = 0 < 1$ and
$1^2 + 2^2 = 5 < 100$.

\textbf{KKT conditions}: Lagrangian:
$L(x_1,x_2,\lambda_1,\lambda_2) = x_1 + x_2 +
\lambda_1((x_1-1)^2 + (x_2-2)^2 - 1) +
\lambda_2(x_1^2 + x_2^2 - 100)$.

For optimal $(x_1^*, x_2^*)$ and $\lambda_1^*, \lambda_2^*$:
\begin{itemize}[nosep]
\item Primal feasibility: $(x_1^*-1)^2 + (x_2^*-2)^2 \leq 1$,
$x_1^{*2} + x_2^{*2} \leq 100$
\item Dual feasibility: $\lambda_1^*, \lambda_2^* \geq 0$
\item Complementary slackness:
$\lambda_1^*((x_1^*-1)^2 + (x_2^*-2)^2 - 1) = 0$,
$\lambda_2^*(x_1^{*2} + x_2^{*2} - 100) = 0$
\item Stationarity: $1 + 2\lambda_1^*(x_1^* - 1) +
2\lambda_2^* x_1^* = 0$, $1 + 2\lambda_1^*(x_2^* - 2) +
2\lambda_2^* x_2^* = 0$
\end{itemize}

\textbf{Verify global minimum}: For
$(x_1^*, x_2^*) = (1 - \frac{\sqrt{2}}{2}, 2 -
\frac{\sqrt{2}}{2})$, try $\lambda_1^* = \frac{1}{\sqrt{2}}$,
$\lambda_2^* = 0$. From complementary slackness,
$\lambda_2^* = 0$ implies second constraint inactive. From
stationarity, $\lambda_1^* = \frac{1}{\sqrt{2}}$ satisfies
equations. All KKT conditions hold, so this is a global
minimum with optimal value $3 - \sqrt{2}$.

\textbf{Sensitivity analysis}: Perturbed problem
$\min x_1 + x_2$ s.t. $(x_1-1)^2 + (x_2-2)^2 + 2\varepsilon
\leq 1$, $x_1^2 + x_2^2 - 5\varepsilon \leq 100$. Let
$P^*(\varepsilon)$ be optimal value. First-order Taylor:
$P^*(\varepsilon) \approx P^*(0) - \lambda_1^*(-2\varepsilon)
- \lambda_2^*(5\varepsilon) = 3 - \sqrt{2} -
\lambda_1^*(-2\varepsilon) = 3 - \sqrt{2} + \sqrt{2}\varepsilon$.
