\begin{homeworkProblem}
  Find all constant coefficients $a_0,a_1,a_2,a_3$ for which the
  following problem is a convex optimization:
  \begin{subequations}
    \begin{align}
      &\min_{x\in\mathbb R^2}\  \
      \quad \left(\frac{1}{12} x_1^4+a_3x_1^3+a_2x_1^2+a_1x_1+a_0\right)
      + x_2^{18} \\
      &\text{ s.t.}\qquad a_2x_1^2+a_1x_1+a_0\leq -10x_2+5\\
      &\hspace{1.2cm} a_1x_1+a_0=x_2+1
    \end{align}
  \end{subequations}
\end{homeworkProblem}
