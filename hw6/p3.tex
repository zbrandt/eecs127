\begin{homeworkProblem}
    Consider a convex function $f(x_1,x_2):\mathbb R^2\rightarrow
    \mathbb R$ such that:
    \begin{itemize}
        \item The points $(2,2)$ and $(-2,4)$ are both global minima of
            the function $f(x_1,x_2)$ with the property that $f(2,2)=5$.
        \item $f(0,0)$ is equal to 10.
    \end{itemize}

    Answer the following questions:
    \begin{itemize}
        \item [i)] Prove that the point $(0,3)$ is a global minimum of
            the function $f(x_1,x_2)$.
        \item [ii)] Prove that the point $(0,-1)$ is not a local minimum
            of the function $f(x_1,x_2)$.
    \end{itemize}

    \begin{solution}
        \begin{itemize}
            \item[i)] The point $(0,3)$ lies on the line segment between the 
                points $(2,2)$ and $(-2,4)$ when $\alpha = \frac{1}{2}$. Since 
                $f(x_1,x_2)$ is convex, for any two points $x,y \in \mathbb 
                R^2$ and any $\alpha \in [0,1]$, $f(\alpha x + (1-\alpha)y) 
                \leq \alpha f(x) + (1-\alpha)f(y)$. Substituting in $x = (2,2)$
                $y = (-2,4)$ and $\alpha = \frac{1}{2}$ into the inequality,
                \[
                    \begin{split}
                        f(\alpha(2,2) + (1-\alpha)(-2,4)) &\leq \frac{1}{2}f(2,2) + \frac{1}{2}f(-2,4) \\
                        f(0,3) &\leq \frac{1}{2}(5) + \frac{1}{2}(5) \\
                        f(0,3) &\leq 5
                    \end{split}
                \]

                Since $f(2,2) = f(-2,4) = 5$ are both global minima, $f(0,3) = 
                5$ and is a global minimum. 

            \item[ii)] There exists a line through the points $(0,-1)$ and 
                $(0,3)$ that goes through the point $(0,0)$. When $\alpha$ is
                equal to $\frac{3}{4}$, the point is on the line $\alpha(0,3) + 
                (1-\alpha)(0,-1) = (0,0)$. Since $f(x_1, x)_2$ is a convex
                function,
                \[
                    \begin{split}
                        f(0,0) &\leq \frac{3}{4}f(0,-1) + \frac{1}{4}f(0,3) \\
                        10 &\leq \frac{3}{4}f(0,-1) + \frac{1}{4}(5) \\
                        \frac{35}{4} &\leq  f(0, -1)
                    \end{split}
                \]

                So, $f(0,-1)$ is at least $\frac{35}{4}$, which is greater than
                $f(0,3)$. Therefore, $(0,-1)$ cannot be a local minimum of 
                $f(x_1,x_2)$, because for any $\alpha \in [0,1]$
                \[
                    f(\alpha(0,-1) + (1-\alpha)(0,3)) \leq \alpha f(0,-1) + 
                    (1-\alpha)f(0,3) \leq f(0,-1)
                \]

                So any point on the line segment between $(0,-1)$ and $(0,3)$,
                which lies at a global minimum, has a function value less than
                that at $(0,-1)$. Any neighborhood around $(0,-1)$ will contain
                points on this line segment, so $(0,-1)$ cannot be a local
                minimum.

        \end{itemize}
    \end{solution}

\end{homeworkProblem}
