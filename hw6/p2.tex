\begin{homeworkProblem}
  Consider the optimization problem: 
  \begin{subequations}
    \begin{align}
      &\min_{x\in\mathbb R^3}\  \
      \quad e^{-x_1-x_2-x_3}+(x_1+2x_2+3x_3)^4 \\
      &\text{ s.t.}\qquad x_1-x_2-x_3=1 \\
      &\qquad\quad \ \  x_1^2+x_2^2+x_3^2+x_1x_2+x_2x_3+x_1x_3\leq 2\\
      &\qquad\quad \ \  0\leq  x_1\leq 1 \\
      &\qquad\quad \ \   0\leq  x_2\leq 1  \\
      &\qquad\quad \ \   0\leq  x_3\leq 1  
    \end{align}
  \end{subequations}
  \begin{itemize}
    \item [i)] Prove that $x=[1 \  \ 0 \  \ 0]^T$ is a feasible
    point.

    \item [ii)] Using the Weierstrass theorem, prove that the above
    optimization problem has a solution.

    \item [iii)] Prove that every local minimum of the above
    optimization problem is a global minimum.
  \end{itemize}
\end{homeworkProblem}
